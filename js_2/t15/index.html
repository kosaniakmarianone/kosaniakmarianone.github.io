4. Використайте routing від vue.js для створення SPA

Приклад routing:

<script> document.addEventListener('DOMContentLoaded', async function(){ let home = await axios.get("templates/home.html"); let notfound = await axios.get("templates/404.html"); const data = { message: 'Hello Vue.js!', currentPath: window.location.hash, }; const Home = { template: home.data, }; const NotFound = { template: notfound.data }; const routes = { '/': Home, '/home': Home, '/not-found': NotFound, } const app = { data() { return data }, methods: { }, components: { }, computed: { currentView() { return routes[this.currentPath.slice(1) || '/'] || NotFound } }, mounted() { window.addEventListener('hashchange', () => { this.currentPath = window.location.hash }) } } Vue.createApp(app).mount('#app'); }) </script>
